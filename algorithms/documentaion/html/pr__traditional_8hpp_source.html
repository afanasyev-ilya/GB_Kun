53 std::vector<int> incoming_degrees(vertices_count, 0);

55 std::vector<T> old_page_ranks(vertices_count, 0);
56
57 // init ranks and other data
58 for(int i = 0; i < vertices_count; i++)
59 {
60 ranks[i] = 1.0/((T)vertices_count);
61 number_of_loops[i] = 0;
62 }
63
64 // calculate number of loops
65 for(int src_id = 0; src_id < vertices_count; src_id++)
66 {
67 Index shift = graph->get_row_ptr()[src_id];
68 Index connections_count = graph->get_row_ptr()[src_id + 1] - graph->get_row_ptr()[src_id];
69 for(Index edge_pos = 0; edge_pos < connections_count; edge_pos++)
70 {
71 Index dst_id = graph->get_col_ids()[shift + edge_pos];

74 }
75 }
76

81 }
82

84 for(int i = 0; i < vertices_count; i++)

87 }
