45void LAGraph_page_rank_sinks (GrB_Vector centrality, // centrality(i): GAP-style pagerank of node i
46 // inputs:
47 LAGraph_Graph<float> *G, // input graph
48 int *iters, // output: number of iterations taken
49 int itermax = 100, // maximum number of iterations (typically 100)
50 double damping = 0.85, // damping factor (typically 0.85)
51 double tol = 1e-4 // stopping tolerance (typically 1e-4) ;
52)
53{
54 GrB_Matrix AT = G->AT;
55 lablas::Vector<Index>* d_out = G->coldegree; // TODO row degree and transpose
56 GrB_Vector r = NULL, *d = NULL, *t = NULL, *w = NULL, *d1 = NULL ;
57 lablas::Descriptor desc;
58
59 //--------------------------------------------------------------------------
60 // initializations
61 //--------------------------------------------------------------------------
62
63 GrB_Index n ;
64 GrB_TRY (GrB_Matrix_nrows (&n, AT)) ;
65
66 const double scaled_damping = (1 - damping) / n ;
67 float rdiff = 1 ; // first iteration is always done
68
69 // r = 1 / n
70 GrB_TRY (GrB_Vector_new (&t, GrB_FP32, n)) ;
71 GrB_TRY (GrB_Vector_new (&r, GrB_FP32, n)) ;
72 GrB_TRY (GrB_Vector_new (&w, GrB_FP32, n)) ;
73 GrB_TRY (GrB_Vector_new (&d, GrB_FP32, n)) ;
74 GrB_TRY (GrB_Vector_new (&d1, GrB_FP32, n)) ;
75 t->set_name("t");
76 r->set_name("r");
77 w->set_name("w");
78 d->set_name("d");
79 d1->set_name("d1");
80
81 double init_rank = 1.0/n;
82 GrB_TRY (GrB_assign (r, MASK_NULL, NULL, init_rank, GrB_ALL, n, NULL)) ;
83
84 // find all sinks, where sink(i) = true if node i has d_out(i)=0, or with
85 // d_out(i) not present. LAGraph_Property_RowDegree computes d_out =
86 // G->rowdegree so that it has no explicit zeros, so a structural mask can
87 // be used here.
88 lablas::Vector<bool>* sink = NULL;
89 lablas::Vector<float>* rsink = NULL;
90 GrB_Index nsinks, nvals ;
91 GrB_TRY (GrB_Vector_nvals (&nvals, d_out)) ;
92 nsinks = n - nvals ;
93 if (nsinks > 0)
94 {
95 // sink<!struct(d_out)> = true
96 GrB_TRY (GrB_Vector_new (&sink, GrB_BOOL, n)) ;
97 GrB_TRY (GrB_assign (sink, d_out, NULL, (bool)true, GrB_ALL, n, GrB_DESC_SC)) ;
98 GrB_TRY (GrB_Vector_new (&rsink, GrB_FP32, n));
99 sink->set_name("sink");
100 rsink->set_name("rsink");
101 }
102
103 // prescale with damping factor, so it isn't done each iteration
104 // d = d_out / damping ;
105 GrB_TRY (GrB_apply (d, MASK_NULL, NULL, GrB_DIV_FP32, d_out, damping, &desc)) ;
106
107 // d1 = 1 / damping
108 float dmin = 1.0 / damping ;
109 GrB_TRY (GrB_assign (d1, MASK_NULL, NULL, dmin, GrB_ALL, n, NULL)) ;
110
111 // d = max (d1, d)
112 GrB_TRY (GrB_eWiseAdd (d, MASK_NULL, NULL, GrB_MAX_FP32, d1, d, NULL)) ;
113
114 //--------------------------------------------------------------------------
115 // pagerank iterations
116 //--------------------------------------------------------------------------
117 for ((*iters) = 0 ; (*iters) < itermax; (*iters)++)
118 {
119 float teleport = scaled_damping ; // teleport = (1 - damping) / n
120 if (nsinks > 0)
121 {
122 const float damping_over_n = damping / n ;
123 // handle the sinks: teleport += (damping/n) * sum (r (sink))
124 // rsink<struct(sink)> = r
125 GrB_TRY (GrB_Vector_clear (rsink)) ;
126
127 GrB_TRY (GrB_assign (rsink, sink, NULL, r, GrB_ALL, n, GrB_DESC_S));
128
129 // sum_rsink = sum (rsink)
130 float sum_rsink = 0 ;
131 GrB_TRY (GrB_reduce (&sum_rsink, NULL, GrB_PLUS_MONOID_FP32, rsink, NULL)) ;
132 teleport += damping_over_n * sum_rsink ;
133 }
134
135 // swap t and r ; now t is the old score
136 GrB_Vector temp = t ; t = r ; r = temp ;
137 // w = t ./ d
138
139 GrB_TRY (GrB_eWiseMult (w, MASK_NULL, NULL, GrB_DIV_FP32, t, d, NULL)) ;
140
141 // r = teleport
142 GrB_TRY (GrB_assign (r, MASK_NULL, NULL, teleport, GrB_ALL, n, NULL)) ;
143
144 // r += A'*w
145 SAVE_STATS((GrB_TRY (GrB_mxv (r, MASK_NULL, GrB_PLUS_FP32, LAGraph_plus_second_fp32, AT, w, &desc))),
146 "pr_mxv", (sizeof(float)*2 + sizeof(size_t)), 1, AT);
147
148 // t -= r
149 GrB_TRY (GrB_assign (t, MASK_NULL, GrB_MINUS_FP32, r, GrB_ALL, n, NULL)) ;
150 // t = abs (t)
151 GrB_TRY (GrB_apply (t, MASK_NULL, NULL, GrB_ABS_FP32, t, NULL)) ;
152 // rdiff = sum (t)
153 GrB_TRY (GrB_reduce (&rdiff, NULL, GrB_PLUS_MONOID_FP32, t, NULL)) ;
154
155 //float ranks_sum = 0;
156 double ranks_sum = 0;
157 GrB_TRY (GrB_reduce (&ranks_sum, NULL, GrB_PLUS_MONOID_FP32, r, NULL));
158 #ifdef __DEBUG_INFO__
159 cout << "ranks sum: " << ranks_sum << endl;
160 #endif
161 }
162
163 //--------------------------------------------------------------------------
164 // free workspace and return result
165 //--------------------------------------------------------------------------
166
167 centrality->dup(r);
168 GrB_free (&d1) ;
169 GrB_free (&d);
170 GrB_free (&t);
171 GrB_free (&w);
172 GrB_free (&r);
173 if (nsinks > 0)
174 {
175 GrB_free (&sink);
176 GrB_free (&rsink);
177 }
178}
